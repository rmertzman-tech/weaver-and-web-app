<!--

Chosen Palette: Calm Neutrals

Application Structure Plan: A single-page, narrative scrolling application that follows the 'Weaver and the Web' analogy, updated to reflect the 'Bounded Weaver' concept. It uses interactive diagrams and a core simulation where users select a life 'domain' before facing a challenge. This makes the distinction between local coherence (TCF) and global weaving skill (ATCF) tangible and interactive. The goal is to guide the user from understanding the components of ethical coherence to experiencing the dynamic process of maintaining it across different life contexts.

Visualization & Content Choices: Report Info: Identity Kernel -> Goal: Define -> Viz: Interactive diagram. Report Info: PRF -> Goal: Explain -> Viz: Clickable diagram with domain-specific examples. Report Info: Web Repair -> Goal: Simulate -> Viz: CSS/JS animated web + domain selector -> Interaction: User selects a domain, triggers damage/repair in that specific domain. Report Info: TCF vs. ATCF -> Goal: Differentiate -> Viz: TCF chart for local/domain coherence, ATCF multi-bar chart for global skill -> Interaction: Charts update based on the selected domain in the simulation. -> Justification: This interactive structure makes the abstract difference between a static state (TCF) and a dynamic capability (ATCF) clear, and shows how resilience is built across multiple life domains. -> Library: Chart.js.

CONFIRMATION: NO SVG graphics used. NO Mermaid JS used.

const prfDetails = {

'Beliefs': {

main: "Your map of the world, which tells you where the web needs strengthening. These are your assumptions about reality, causality, and human nature.",

domain: "For example, your beliefs about what makes a 'good relationship' are different from your beliefs about what makes a 'successful career'."

},

'Rules': {

main: "Your instincts and strategies for how to repair a tear or incorporate a new anchor point. These can be conscious principles or unconscious habits learned over time.",

domain: "The 'rules' you use to resolve a conflict with a loved one (e.g., prioritize empathy) are likely different from the rules you use to complete a work project (e.g., prioritize efficiency)."

},

'Ontology': {

main: "Your way of categorizing the world, which determines whether you perceive an event as a threat, an opportunity, or irrelevant noise.",

domain: "A challenging conversation might be categorized as a 'threat' in a work context but an 'opportunity for growth' in a relationship."

},

'Authenticity': {

main: "This is the crucial feedback loopthe spider's 'felt sense' of the web's integrity. When the web is taut and functional, you feel a sense of 'flow' or coherence.",

domain: "The feeling of authenticity at work (e.g., competence, purpose) might feel different from authenticity in your personal growth (e.g., vulnerability, learning)."

}

};

 

window.showPrfDetail = (key) => {

document.getElementById('modal-title').innerText = key;

document.getElementById('modal-text-main').innerText = prfDetails[key].main;

document.getElementById('modal-text-domain').innerText = prfDetails[key].domain;

document.getElementById('modal-overlay').classList.remove('hidden');

};

 

window.closeModal = () => {

document.getElementById('modal-overlay').classList.add('hidden');

};

 

// Kernel Values Interaction

const kernelInputsContainer = document.getElementById('kernel-inputs');

const kernelValuesContainer = document.getElementById('kernel-values');

for (let i = 0; i < 5; i++) {

const input = document.createElement('input');

input.type = 'text';

input.placeholder = `Value ${i + 1}`;

input.className = 'border border-stone-300 rounded-md px-2 py-1 text-sm w-28 focus:ring-2 focus:ring-purple-400 focus:border-purple-400';

input.addEventListener('input', updateKernelVisual);

kernelInputsContainer.appendChild(input);

}

 

function updateKernelVisual() {

kernelValuesContainer.innerHTML = '';

const values = Array.from(kernelInputsContainer.querySelectorAll('input')).map(input => input.value).filter(Boolean);

values.forEach((value, i) => {

const angle = (i / values.length) * 2 * Math.PI;

const x = 50 + 45 * Math.cos(angle);

const y = 50 + 45 * Math.sin(angle);

const valueEl = document.createElement('div');

valueEl.className = 'absolute bg-stone-200 text-stone-700 text-xs font-semibold px-2 py-1 rounded-full shadow';

valueEl.style.left = `${x}%`;

valueEl.style.top = `${y}%`;

valueEl.style.transform = 'translate(-50%, -50%)';

valueEl.innerText = value;

kernelValuesContainer.appendChild(valueEl);

});

}

updateKernelVisual();

 

// Web Visualization

const webViz = document.getElementById('web-viz');

const domains = {

relationships: [0, 1],

career: [2, 3],

growth: [4, 5],

health: [6, 7]

};

const numLines = 8;

for (let i = 0; i < numLines; i++) {

const line = document.createElement('div');

line.className = 'web-line';

line.dataset.lineId = i;

line.style.width = '300px';

line.style.height = '2px';

line.style.top = '149px';

line.style.left = '0px';

line.style.transform = `rotate(${i * (180 / numLines)}deg)`;

webViz.appendChild(line);

}

 

// State Management

let state = {

selectedDomain: null,

domainScores: {

relationships: { tcf: 100, atcf: 70 },

career: { tcf: 100, atcf: 70 },

growth: { tcf: 100, atcf: 70 },

health: { tcf: 100, atcf: 70 }

},

isDamaged: false

};

 

// Chart.js setup

const tcfChartCtx = document.getElementById('tcfChart').getContext('2d');

const tcfChart = new Chart(tcfChartCtx, {

type: 'bar',

data: {

labels: ['Coherence'],

datasets: [{

label: 'TCF Score',

data: [0],

backgroundColor: ['rgba(168, 85, 247, 0.6)'],

borderColor: ['rgba(168, 85, 247, 1)'],

borderWidth: 1

}]

},

options: { maintainAspectRatio: false, scales: { y: { beginAtZero: true, max: 100 } }, plugins: { legend: { display: false } } }

});

 

const atcfChartCtx = document.getElementById('atcfChart').getContext('2d');

const atcfChart = new Chart(atcfChartCtx, {

type: 'bar',

data: {

labels: Object.keys(domains).map(d => d.charAt(0).toUpperCase() + d.slice(1)),

datasets: [{

label: 'ATCF Score',

data: Object.values(state.domainScores).map(d => d.atcf),

backgroundColor: 'rgba(139, 92, 246, 0.6)',

borderColor: 'rgba(139, 92, 246, 1)',

borderWidth: 1

}]

},

options: { maintainAspectRatio: false, scales: { y: { beginAtZero: true, max: 100 } }, plugins: { legend: { display: false } } }

});

 

// Simulation Logic

const domainSelector = document.getElementById('domain-selector');

const damageBtn = document.getElementById('damage-btn');

const repairBtn = document.getElementById('repair-btn');

const tcfDomainLabel = document.getElementById('tcf-domain-label');

 

domainSelector.addEventListener('click', (e) => {

if (e.target.classList.contains('domain-btn')) {

const domain = e.target.dataset.domain;

if (state.isDamaged && domain !== state.selectedDomain) return;

state.selectedDomain = domain;

domainSelector.querySelectorAll('.domain-btn').forEach(btn => btn.classList.remove('active'));

e.target.classList.add('active');

updateWebVisual();

updateCharts();

damageBtn.disabled = state.isDamaged;

}

});

 

function updateWebVisual() {

const allLines = webViz.querySelectorAll('.web-line');

allLines.forEach(line => {

line.classList.remove('active', 'damaged');

line.style.transform = `rotate(${line.dataset.lineId * (180 / numLines)}deg) scaleX(1)`;

});

 

if (state.selectedDomain) {

const activeLines = domains[state.selectedDomain];

activeLines.forEach(id => {

const line = webViz.querySelector(`[data-line-id='${id}']`);

if (line) line.classList.add('active');

});

}

}

 

function updateCharts() {

if (state.selectedDomain) {

const score = state.domainScores[state.selectedDomain].tcf;

tcfChart.data.datasets[0].data[0] = score;

tcfDomainLabel.innerText = `Current Focus: ${state.selectedDomain.charAt(0).toUpperCase() + state.selectedDomain.slice(1)}`;

} else {

tcfChart.data.datasets[0].data[0] = 0;

tcfDomainLabel.innerText = 'Select a domain to see its coherence.';

}

tcfChart.update();

 

atcfChart.data.datasets[0].data = Object.values(state.domainScores).map(d => d.atcf);

atcfChart.update();

}

 

damageBtn.addEventListener('click', () => {

if (!state.selectedDomain) return;

state.isDamaged = true;

state.domainScores[state.selectedDomain].tcf = Math.max(0, state.domainScores[state.selectedDomain].tcf - Math.floor(Math.random() * 30 + 20));

const activeLines = domains[state.selectedDomain];

activeLines.forEach(id => {

const line = webViz.querySelector(`[data-line-id='${id}']`);

if (line) {

line.classList.add('damaged');

line.style.transform = `rotate(${line.dataset.lineId * (180 / numLines)}deg) scaleX(${Math.random() * 0.4 + 0.3})`;

}

});

 

updateCharts();

damageBtn.disabled = true;

repairBtn.disabled = false;

});

 

repairBtn.addEventListener('click', () => {

if (!state.selectedDomain) return;

state.isDamaged = false;

state.domainScores[state.selectedDomain].tcf = 100;

state.domainScores[state.selectedDomain].atcf = Math.min(100, state.domainScores[state.selectedDomain].atcf + Math.floor(Math.random() * 5 + 5));

 

updateWebVisual();

updateCharts();

damageBtn.disabled = false;

repairBtn.disabled = true;

});

});
